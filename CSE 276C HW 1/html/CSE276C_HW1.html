% Square and Full Rank Matrix A.

% Run the PA = LDU Decomposition User-defined Function

% Display Lower Triangular Matrix with 1 along the diagonal
fprintf('\nL = \n\n')

% Display Diagonal Matrix
fprintf('\nD = \n\n')
disp(D)

% Display Upper Triangular Matrix with 1 along the diagonal
fprintf('\nU = \n\n')
disp(U)

% Display PA
fprintf('\nPA = \n\n')
disp(P*A)

% Display LDU
fprintf('\nLDU = \n\n')
disp(L*D*U)
